# cli-audiosytem
## Running
./cli-audiosystem [dir] [audio info]

[dir]
- -p Playback
- -c Capture

[audio info]
- -r=samplerate
- -b=bitformat
- -l=channels

## Available Bit Formats 
s16le = Signed 16 bit Little Endian 
s16be = Signed 16 bit Big Endian 
u16le = Unsigned 16 bit Little Endian 
u16be = Unsigned 16 bit Big Endian
